(function () {
  'use strict';

  class Organization {
    constructor(name) {
      this.name = name;
    }
    get nameUpperCase() {
      return this.name.toUpperCase();
    }
  }

  class Company extends Organization {
    constructor(name) {
      super(name);
    }
    get nameUpperCase() {
      return 'Company: ' + super.nameUpperCase;
    }
  }

  var pixelgrid = new Company('PixelGrid');
  console.log(pixelgrid.nameUpperCase);

})();
