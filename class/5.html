(function () {
  'use strict';

  class Organization {
    constructor(name) {
      this.name = name;
    }
    greet() {
      return "Hello! We are " + this.name + "!!";
    }
  }

  class Company extends Organization {
    constructor(name, location) {
      super(name);
      this.location = location;
    }
    greet() {
      return super.greet() + " We are based in " + this.location + ".";
    }
  }

  var pixelgrid = new Company("PixelGrid", "Tokyo");
  print(pixelgrid.name);
  print(pixelgrid.location);

})();
